Tips et méthodes utiles

              
                h = { foo: 'bar' }
                h[:foo]            # => "bar"
                h.empty?           # => false
                h.has_key? 'nope'  # => false
                h.has_value? 'bar' # => true
                h.keys             # => [:foo]

                h = { first: 'one', second: 'two' }
                h.reject! { |k, v| k == :first }
                h # => {:second=>"two"}


                does = is = { true => 'Yes', false => 'No' }
                does[10 == 50] # => "No"
                is[10 > 5]     # => "Yes"
